This example interacts with two Localfocus charts. Both charts have interactive features (age groups). The regular interaction interface for users has been disabled in the left chart.

Working example code

                
<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="utf-8">
        <meta http-equiv="X-UA-Compatible" content="IE=edge">
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <title>Simple interaction</title>
        <link href="../libs/bootstrap/css/bootstrap.min.css" rel="stylesheet">
    </head>
    <body>
        <div class="container" style="text-align:center">
            <h1>Simple Interaction</h1>
            <p>Select your age group. Last selected is <strong class="current">15 - 75 years</strong></p>
            <div class="well">
                <h4>Adjust both charts:</h4>
                <button class="btn btn-default">15 - 75 years</button>
                <button class="btn btn-default">15 - 25 years</button>
                <button class="btn btn-default">25 - 45 years</button>
                <button class="btn btn-default">45 - 75 years</button>

            </div>
            <div class="row">
                <div class="col-md-6">
                    <div class="well">
                        <h4>Adjust left chart only:</h4>
                        <button class="btn btn-default" data-target="#left">15 - 75 years</button>
                        <button class="btn btn-default" data-target="#left">15 - 25 years</button>
                        <button class="btn btn-default" data-target="#left">25 - 45 years</button>
                        <button class="btn btn-default" data-target="#left">45 - 75 years</button>
                    </div>
                    <iframe class="localfocusvisual" id="left" scrolling="no" frameborder="0" style="width:100%;height:500px;overflow:hidden" src="https://localfocus2.appspot.com/5614d60becd61?api=1&noui=1"></iframe>
                </div>
                <div class="col-md-6">
                    <div class="well">
                        <h4>Adjust right chart only:</h4>
                        <button class="btn btn-default" data-target="#right">15 - 75 years</button>
                        <button class="btn btn-default" data-target="#right">15 - 25 years</button>
                        <button class="btn btn-default" data-target="#right">25 - 45 years</button>
                        <button class="btn btn-default" data-target="#right">45 - 75 years</button>
                    </div>
                    <iframe class="localfocusvisual" id="right" scrolling="no" frameborder="0" style="width:100%;height:500px;overflow:hidden" src="http://localfocus2.appspot.com/5614d6776129c?api=1&noui=1"></iframe>
                </div>
            </div>
        </div>
        <script src="../libs/jquery/jquery-1.11.2.min.js"></script>
        <script src="../libs/bootstrap/js/bootstrap.min.js"></script>
        <script src="../localfocusapi.js"></script>
        <script>
            $(function(){ // DOM is ready
                $('button').click(function(){ // Find all buttons and add click handler
                    var elem = $(this); 
                    var label = elem.text(); // Get the button's label
                    var target = elem.attr('data-target'); // Get the iframe's #id
                    if(target){
                        // Only change left or right widget
                        LocalFocusAPI.select(target).activate(label);
                    } else {
                        // Change both widgets
                        LocalFocusAPI.selectAll('.localfocusvisual',function(){
                            this.activate(label);
                        });
                    }

                    $('strong.current').text(label);
                });
            });
        </script>
    </body>
</html>
